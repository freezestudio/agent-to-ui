# A2UI v1.0 协议实现计划

> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 在 a2a-dev monorepo 中实现 A2UI v1.0 协议，包括核心类型、框架无关渲染器、A2A 扩展、Angular v22 前端应用和示例 Agent。

**架构：** 方案 C（混合优化）——Agent 生成 A2UI 消息打包为 A2A DataPart → 客户端提取 → MessageProcessor 解析 → SurfaceModel 状态更新 → Angular 渲染。

**技术栈：** TypeScript, Zod v4, A2A v1.0 (已有), Vite+, Angular v22, markdown-it, DOMPurify, Vitest

**协议资源：** `a2ui-protocol/specification/v1_0/`（catalogs, JSON schemas, 36 个示例文件）

```ts
import { z } from "zod";

export const ComponentIdSchema = z.string().min(1);

export const DataBindingSchema = z.object({
  path: z.string().min(1),
}).strict();

export const FunctionCallSchema = z.object({
  call: z.string().min(1),
  args: z.record(z.string(), z.unknown()).optional(),
});

export const DynamicStringSchema = z.union([z.string(), DataBindingSchema, FunctionCallSchema]);
export const DynamicNumberSchema = z.union([z.number(), DataBindingSchema, FunctionCallSchema]);
export const DynamicBooleanSchema = z.union([z.boolean(), DataBindingSchema, FunctionCallSchema]);
export const DynamicStringListSchema = z.union([
  z.array(z.string()), DataBindingSchema, FunctionCallSchema,
]);

export const ComponentCommonSchema = z.object({
  id: z.string().min(1),
  accessibility: z.object({
    label: DynamicStringSchema.optional(),
    description: DynamicStringSchema.optional(),
  }).optional(),
});

export const ChildListSchema: z.ZodType<unknown> = z.union([
  z.array(z.string()),
  z.object({ componentId: z.string(), path: z.string() }).strict(),
]);

export const CheckRuleSchema = z.object({
  condition: DynamicBooleanSchema,
  message: z.string(),
});
```

```ts
  surfaceId: z.string().min(1),
  path: z.string().optional(),
  value: z.unknown().optional(),

export const DeleteSurfaceSchema = z.object({
  surfaceId: z.string().min(1),
}).strict();

export const ActionResponseSchema = z.object({
  actionId: z.string().min(1),

}).strict();

export const A2uiMessageSchema = z.union([
  z.object({ version: z.literal(SPEC_VERSION), createSurface: CreateSurfaceSchema }),

  z.object({ version: z.literal(SPEC_VERSION), deleteSurface: DeleteSurfaceSchema }),
  z.object({ version: z.literal(SPEC_VERSION), actionResponse: ActionResponseSchema }),
```

```ts
export type EventHandler<T> = (data: T) => void;

export class EventSource<T> {
  private handlers = new Set<EventHandler<T>>();

  subscribe(handler: EventHandler<T>): () => void {
    this.handlers.add(handler);
    return () => this.handlers.delete(handler);
  }

  emit(data: T): void {
    for (const handler of this.handlers) {
      handler(data);
    }
  }

  dispose(): void { this.handlers.clear(); }
}
```

```ts
import { EventSource } from "../common/events.js";

export class DataModel {
  private root: unknown = {};
  readonly onUpdated = new EventSource<{ path: string; value: unknown }>();

  resolve(path: string): unknown {
    if (path === "/" || !path) return this.root;
    const parts = path.split("/").filter(Boolean);
    let current: unknown = this.root;
    for (const part of parts) {
      if (current && typeof current === "object") {
        current = (current as Record<string, unknown>)[part];
      } else return undefined;
    }
    return current;
  }

  set(path: string, value: unknown): void {
    if (path === "/" || !path) {
      this.root = value;
    } else {
      const parts = path.split("/").filter(Boolean);
      let current: Record<string, unknown> = this.root as Record<string, unknown>;
      for (let i = 0; i < parts.length - 1; i++) {
        if (!current[parts[i]] || typeof current[parts[i]] !== "object") {
          current[parts[i]] = {};
        }
        current = current[parts[i]] as Record<string, unknown>;
      }
      if (value === undefined) {
        delete current[parts[parts.length - 1]];
      } else {
        current[parts[parts.length - 1]] = value;
      }
    }
    this.onUpdated.emit({ path, value });
  }

  getSnapshot(): Record<string, unknown> {
    return JSON.parse(JSON.stringify(this.root)) as Record<string, unknown>;
  }
}
```

```ts
export class SurfaceComponentsModel {

  readonly onCreated = new EventSource<ComponentModel>();
  readonly onDeleted = new EventSource<string>();

  get(id: string): ComponentModel | undefined { return this.components.get(id); }
  getAll(): Map<string, ComponentModel> { return this.components; }

  addOrUpdate(id: string, type: string, properties: Record<string, unknown>): void {

      this.components.get(id)!.setProperties(properties);
    } else {
      const model = new ComponentModel(id, type, properties);
      this.components.set(id, model);
      this.onCreated.emit(model);

    if (comp) { comp.dispose(); this.components.delete(id); this.onDeleted.emit(id); }
  }

  dispose(): void { for (const c of this.components.values()) c.dispose(); this.components.clear(); }
```

```ts
export class SurfaceGroupModel {
  private surfaces = new Map<string, SurfaceModel>();
  readonly onSurfaceCreated = new EventSource<SurfaceModel>();
  readonly onSurfaceDeleted = new EventSource<string>();
  readonly onAction = new EventSource<A2uiClientAction>();
  private subs = new Map<string, () => void>();

  getSurface(id: string): SurfaceModel | undefined { return this.surfaces.get(id); }
  getAllSurfaces(): SurfaceModel[] { return Array.from(this.surfaces.values()); }

  addSurface(surface: SurfaceModel): void {
    if (this.surfaces.has(surface.id)) return;
    this.surfaces.set(surface.id, surface);
    const unsub = surface.onAction.subscribe((a) => this.onAction.emit(a));
    this.subs.set(surface.id, unsub);
    this.onSurfaceCreated.emit(surface);
  }

  deleteSurface(id: string): void {
    const surface = this.surfaces.get(id);
    if (surface) {
      this.subs.get(id)?.();
      this.subs.delete(id);
      surface.dispose();
      this.surfaces.delete(id);
      this.onSurfaceDeleted.emit(id);
    }
  }

  dispose(): void {
    for (const id of [...this.surfaces.keys()]) this.deleteSurface(id);
  }
```

```ts
import { DynamicStringSchema, DataBindingSchema } from "@a2a-dev/a2ui-core";
import { DataModel } from "../state/data-model.js";

export type FunctionImpl = (args: Record<string, unknown>, dataModel?: DataModel) => unknown | Promise<unknown>;

export class FunctionInvoker {
  private functions = new Map<string, FunctionImpl>();
  private dataModel?: DataModel;

  constructor(dataModel?: DataModel) { this.dataModel = dataModel; }

  register(name: string, impl: FunctionImpl): void {
    this.functions.set(name, impl);
  }

  async call(callArgs: { call: string; args?: Record<string, unknown> }): Promise<unknown> {
    const fn = this.functions.get(callArgs.call);
    if (!fn) throw new Error(`未知函数: ${callArgs.call}`);
    const resolvedArgs = await this.resolveArgs(callArgs.args ?? {});
    return fn(resolvedArgs, this.dataModel);
  }

  private async resolveArgs(args: Record<string, unknown>): Promise<Record<string, unknown>> {
    const resolved: Record<string, unknown> = {};
    for (const [key, value] of Object.entries(args)) {
      resolved[key] = this.resolveValue(value);
    }
    return resolved;
  }

  private resolveValue(value: unknown): unknown {
    if (value && typeof value === "object" && "path" in value) {
      return this.dataModel?.resolve((value as { path: string }).path);
    }
    if (value && typeof value === "object" && "call" in value) {
      return this.call(value as { call: string; args?: Record<string, unknown> });
    }
    return value;
  }

  /** 注册 14 个内置函数 */
  registerBuiltins(): void {
    this.register("required", (args) => args.value !== null && args.value !== undefined && args.value !== "");
    this.register("regex", (args) => new RegExp(args.pattern as string).test(String(args.value ?? "")));
    this.register("length", (args) => {
      const len = String(args.value ?? "").length;
      if (args.min !== undefined && len < (args.min as number)) return false;
      if (args.max !== undefined && len > (args.max as number)) return false;
      return true;
    });
    this.register("numeric", (args) => {
      const val = Number(args.value);
      if (isNaN(val)) return false;
      if (args.min !== undefined && val < (args.min as number)) return false;
      if (args.max !== undefined && val > (args.max as number)) return false;
      return true;
    });
    this.register("email", (args) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(String(args.value ?? "")));
    this.register("formatString", (args) => {
      let str = String(args.value ?? "");
      str = str.replace(/\$\{(\/[^}]+)\}/g, (_, path) => String(this.dataModel?.resolve(path) ?? ""));
      return str;
    });
    this.register("formatNumber", (args) => {
      const n = Number(args.value);
      const decimals = args.decimals !== undefined ? Number(args.decimals) : undefined;
      return decimals !== undefined ? n.toFixed(decimals) : n.toLocaleString();
    });
    this.register("formatCurrency", (args) => {
      const n = Number(args.value);
      const currency = String(args.currency ?? "USD");
      return n.toLocaleString(undefined, { style: "currency", currency });
    });
    this.register("formatDate", (args) => {
      const d = new Date(String(args.value ?? ""));
      return isNaN(d.getTime()) ? String(args.value) : d.toLocaleDateString();
    });
    this.register("pluralize", (args) => {
      const count = Number(args.value);
      if (count === 0 && args.zero) return args.zero;
      if (count === 1 && args.one) return args.one;
      return args.other ?? "";
    });
    this.register("openUrl", (args) => {
      if (typeof window !== "undefined") window.open(String(args.url), "_blank");
    });
    this.register("and", (args) => (args.values as boolean[]).every(Boolean));
    this.register("or", (args) => (args.values as boolean[]).some(Boolean));
    this.register("not", (args) => !args.value);
  }
}
```

```ts
import type { A2uiMessage, CreateSurfaceMessage, UpdateComponentsMessage, UpdateDataModelMessage, DeleteSurfaceMessage } from "@a2a-dev/a2ui-core";
import { SurfaceGroupModel } from "../state/surface-group-model.js";
import { SurfaceModel } from "../state/surface-model.js";
import type { A2uiClientAction } from "@a2a-dev/a2ui-core";

export class MessageProcessor {
  readonly model: SurfaceGroupModel;

  constructor(private actionHandler?: (action: A2uiClientAction) => void) {
    this.model = new SurfaceGroupModel();
    if (actionHandler) {
      this.model.onAction.subscribe(actionHandler);
    }
  }

  processMessages(messages: A2uiMessage[]): void {
    for (const msg of messages) {
      if ("createSurface" in msg) this.processCreateSurface(msg.createSurface as CreateSurfaceMessage & { components?: unknown[]; dataModel?: Record<string, unknown> });
      else if ("updateComponents" in msg) this.processUpdateComponents(msg.updateComponents as UpdateComponentsMessage);
      else if ("updateDataModel" in msg) this.processUpdateDataModel(msg.updateDataModel as UpdateDataModelMessage);
      else if ("deleteSurface" in msg) this.processDeleteSurface(msg.deleteSurface as DeleteSurfaceMessage);
      // actionResponse 和 callFunction 暂由外层处理
    }
  }

  private processCreateSurface(payload: CreateSurfaceMessage & { components?: unknown[]; dataModel?: Record<string, unknown> }): void {
    const surface = new SurfaceModel(payload.surfaceId, payload.sendDataModel);
    this.model.addSurface(surface);
    if (payload.components) {
      for (const comp of payload.components) {
        const def = comp as { id: string; component: string };
        if (def.id && def.component) {
          const { id, component, ...props } = def;
          surface.componentsModel.addOrUpdate(id, component, props);
        }
      }
    }
    if (payload.dataModel) {
      surface.dataModel.set("/", payload.dataModel);
    }
  }

  private processUpdateComponents(payload: UpdateComponentsMessage): void {
    const surface = this.model.getSurface(payload.surfaceId);
    if (!surface) return;
    for (const comp of payload.components) {
      const { id, component, ...props } = comp as { id: string; component: string } & Record<string, unknown>;
      if (id && component) {
        surface.componentsModel.addOrUpdate(id, component, props);
      }
    }
  }

  private processUpdateDataModel(payload: UpdateDataModelMessage): void {
    const surface = this.model.getSurface(payload.surfaceId);
    if (!surface) return;
    surface.dataModel.set(payload.path ?? "/", payload.value);
  }

  private processDeleteSurface(payload: DeleteSurfaceMessage): void {
    this.model.deleteSurface(payload.surfaceId);
  }
}
```

```ts
import { A2UI_MIME_TYPE } from "./extension.js";
import type { A2uiMessage } from "@a2a-dev/a2ui-core";

interface TaskLike { artifacts?: Array<{ parts?: Array<{ data?: unknown; mediaType?: string }> }>; }

export function extractA2uiFromTask(task: TaskLike): A2uiMessage[] {
  const messages: A2uiMessage[] = [];
  for (const artifact of task.artifacts ?? []) {
    for (const part of artifact.parts ?? []) {
      if (part.mediaType === A2UI_MIME_TYPE && part.data) {
        const data = part.data as { a2uiMessages?: A2uiMessage[] };
        if (data.a2uiMessages) {
          messages.push(...data.a2uiMessages);
        }
      }
    }
  }
  return messages;
}
```

```ts
    });

    eventQueue.push({
      taskId: context.task!.id,
      status: { state: TaskState.TASK_STATE_COMPLETED, timestamp: new Date().toISOString() },
    });
    eventQueue.complete();
  }

  async cancel(context: RequestContext, eventQueue: EventQueue): Promise<void> {
    eventQueue.push({ taskId: context.task?.id ?? "unknown", status: { state: TaskState.TASK_STATE_CANCELED } });
    eventQueue.complete();
  }
}
```

```ts
  try {
    if (req.method === "GET" && req.url === "/.well-known/agent-card.json") {
      res.writeHead(200, { "Content-Type": "application/json" });
      res.end(JSON.stringify(agentCard, null, 2));
      return;

      const body = await readBody(req);
      const { method, params, id } = JSON.parse(body);
      const result = await handler.handleMethod(method, params);
      res.writeHead(200, { "Content-Type": "application/json" });
      res.end(JSON.stringify({ jsonrpc: "2.0", id, result }));
      return;
    }
  } catch (error: any) {
    const httpStatus = error.httpStatus ?? 500;
    res.writeHead(httpStatus, { "Content-Type": "application/json" });
    res.end(JSON.stringify({ jsonrpc: "2.0", id: null, error: { code: error.code ?? -32603, message: error.message } }));
    return;
  }
  res.writeHead(404);
  res.end(JSON.stringify({ error: "未找到" }));
});

function readBody(req: IncomingMessage): Promise<string> {
  return new Promise((resolve, reject) => {
    let body = "";
    req.on("data", (chunk) => (body += chunk));
```

```ts
import { MessageProcessor, SurfaceGroupModel } from "@a2a-dev/a2ui-renderer";

import { Subject } from "rxjs";

@Injectable({ providedIn: "root" })
export class A2uiRendererService implements OnDestroy {
  private processor: MessageProcessor;

  readonly model: SurfaceGroupModel;
  readonly onAction = new Subject<A2uiClientAction>();

  constructor() {
    this.processor = new MessageProcessor((action) => this.onAction.next(action));
    this.model = this.processor.model;

  processMessages(messages: A2uiMessage[]): void {
    this.processor.processMessages(messages);
  }

  getSurface(id: string) {
    return this.model.getSurface(id);
  }

  ngOnDestroy(): void {
    this.model.dispose();
    this.onAction.complete();
  }
}
```
